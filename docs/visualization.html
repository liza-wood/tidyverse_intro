5.2 Challenge

Conditional statements

weather <- weather %>% 
  mutate(penguin_data_year = case_when(
    year %in% 2007:2009 ~ T,
    T ~ F
  ) )
weather %>% 
  filter(year != min(year) & year != max(year)) %>% 
  ggplot(aes(x = year, y = Temperature.Average..C., group = year,
             color = penguin_data_year)) +
  geom_boxplot()
